DO L[i] ← A[p + i] 6. FOR j ← 0 TO n2 7. DO R[j] ← A[q + j+1 ] 8. L[n1 + 1] ← ∞ 9. R[n2 + 1] ← ∞ 10. i ← 0 11. j ← 0 12. FOR k ← p TO r 13. DO IF L[i ] ≤ R[ j] 14. THEN A[k] ← L[i] 15. i ← i + 1 16. ELSE A[k] ← R[j] 17. j ← j + 1
